Comment utiliser Modern Form Validator

// 1. Inclure le script
<script src="js/formValidator.js"></script>

// 2. Initialiser le validateur
const validator = new ModernFormValidator('#myForm', {
    // Options de configuration (facultatif)
    animationDuration: 300,
    errorClass: 'error',
    successClass: 'success',
    onSuccess: function(form) {
        console.log('Formulaire valide!', form);
    }
});

// 3. Ajouter des attributs data-validate sur les champs
<input type="text" name="username"
    data-validate="required,minLength:3,maxLength:20">
